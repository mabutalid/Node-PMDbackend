exports.getApplications = (req,res,next) => {
  let start_date = req.params.start_date ;

  let inputs = req.query.inputs;
  Application.findAll({
      attributes: ['id','first_name','last_name','area_code','status','date_issued','created_by'],
      where: {
          [Op.and]:{
            date_issued:{
